78 template <typename T>
79 void Stack<T>::check_vector_type_matches_type_name() {
80  bool type_correct;
81  if (_type_name == "int") {
82  type_correct = sizeof(T) == sizeof(int); // rules out everything but unsigned, int, float
83  if (type_correct) {
84  T compare_test_lhs = 6.5;
85  int compare_test_rhs = 6;
86  type_correct = (float) compare_test_lhs == (float) compare_test_rhs; // rules out float
87  }
88  if (type_correct) {
89  T negative_test = -5;
90  type_correct = negative_test < 0; // rules out unsigned
91  }
92  } else if (_type_name == "unsigned") {
93  T negative_test = -5;
94  type_correct = negative_test > 0; // T must be unsigned if this is true
95  } else if (_type_name == "float") {
96  type_correct = sizeof(T) == sizeof(float); // rules out double
97  if (type_correct) {
98  T float_test_lhs = 6.5f;
99  float float_test_rhs = 6.5f;
100  type_correct = (float) float_test_lhs == float_test_rhs; // rules out all int types
101  }
102  } else { // _type_name = "double"
103  type_correct = sizeof(T) == sizeof(double); // leaves only double, long long int and unsigned long long int
104  if (type_correct) {
105  T double_test_lhs = 6.5;
106  double double_test_rhs = 6.5;
107  type_correct = (double) double_test_lhs == double_test_rhs; // rules out all int types
108  }
109  }
110  if (!type_correct) {
111  _values.clear();
112  _calculation.clear();
113  throw TypesNotMatchingException();
114  }
115 }

141 template <typename T>
142 void Stack<T>::remove_unnecessary_whitespace() {
143  if (!_calculation.empty()) {
144  size_t position = 0;
145  // npos is the largest possible size_t. It's the return value of find_first_of() if no occurrence is found
146  while (position != std::string::npos) {
147  position = _calculation.find_first_of("\n\t\r");
148  if (position != std::string::npos) {
149  _calculation.replace(position, 1, " "); // first, replaces all non-space whitespace by spaces
150  }
151  }
152  // removes all non-single spaces
153  std::string::iterator new_end = std::unique(_calculation.begin(), _calculation.end(), both_are_spaces);
154  _calculation.erase(new_end, _calculation.end()); // shortens the string
155  _calculation.erase(_calculation.find_last_not_of(' ') + 1); // crops whitespace at the end of _calculation
156  if (_calculation.find_first_not_of(' ') - 1 != std::string::npos) {
157  // finally crops whitespace at the beginning if necessary
158  _calculation.erase(_calculation.find_first_not_of(' ') - 1, 1);
159  }
160  }
161 }

192 template <typename T>
193 bool Stack<T>::is_correct_operand(std::string& s) {
194  bool correct_operand = true;
195  int num_of_decimal_points = 0;
196  if (_type_name == "int" || _type_name == "unsigned") {
197  for (int i = 0; i < s.size(); i++) {
198  if (s[i] > '9' || s[i] < '0') {
199  correct_operand = false;
200  }
201  if (i == 0 && s[i] == '-' && _type_name == "int" && s.size() > 1) {
202  correct_operand = true;
203  }
204  }
205  } else {
206  for (int i = 0; i < s.size(); i++) {
207  if ((s[i] > '9' || s[i] < '0') && s[i] != '.') {
208  correct_operand = false;
209  }
210  if (i == 0 && s[i] == '-' && s.size() > 1) {
211  correct_operand = true;
212  }
213  if (s[i] == '.') {
214  num_of_decimal_points++;
215  }
216  }
217  if (num_of_decimal_points > 1) {
218  correct_operand = false;
219  }
220  }
221  return correct_operand;
222 }
